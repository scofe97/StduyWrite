## 🧱 왜 이렇게 나누는가?

- 모듈은 **조립의 단위**이며
- 컴포넌트는 **도메인의 본질적 행위 단위**입니다.
- 서비스가 도메인 로직과 조합 로직을 모두 담당하면, **SRP를 위반**하고 **테스트/재사용성**이 떨어집니다.

---

## 📁 패키지 구조 예시

```
├── module                 (💡 비즈니스 흐름 조합을 담당하는 모듈 계층)
│   └── TicketFlowModule.java     // 티켓 시작, 워크플로우 거부 등 복합 절차 담당

├── component              (🔧 도메인 단위의 컴포넌트, 하나의 책임/행위를 가짐)
│   ├── TicketComponent.java         // 티켓 관련 로직 처리 (예: 시작, 초기화)
│   └── WorkflowComponent.java      // 워크플로우 관련 로직 처리 (예: 생성, 상태 이동)

├── handler                (🧠 이벤트 정책 분기 및 전략 처리 핸들러)
│   ├── WorkflowEventHandler.java   // 이벤트 수신 및 정책별 처리 위임
│   ├── ResetHandler.java           // AfterEventPolicy.RESET 처리 전략
│   ├── ProceedHandler.java         // AfterEventPolicy.PROCEED 처리 전략
│   └── AlertHandler.java           // AfterEventPolicy.ALERT 처리 전략

├── event                  (📢 도메인 이벤트 정의 영역)
│   ├── WorkflowRejectedEvent.java  // 워크플로우 거절 이벤트
│   └── TicketStartedEvent.java     // 티켓 시작 이벤트

├── log                    (📝 로깅 및 이력 기록 책임 계층)
│   └── HistoryLogger.java          // 주요 이벤트 로깅, 감사 목적의 추적

├── controller             (🌐 외부 진입을 담당하는 API 레이어)
│   └── TicketController.java       // 티켓 흐름 관련 REST API 정의

└── repository             (💽 데이터베이스 접근 및 MyBatis Mapper 정의)
    ├── TicketMapper.java           // 티켓 관련 DB 작업
    └── WorkflowMapper.java        // 워크플로우 관련 DB 작업
```

---

이제 도메인 로직, 전략, 조합을 완벽히 분리하여

**테스트 가능한 유연한 설계**가 되었습니다.

이제 이 구조에 기반하여 이벤트 스토어, 실패 보상 전략, 상태머신 등도 덧붙일 수 있습니다.

```java
// 예제 구조 개편 버전
// TicketFlowService를 모듈 역할로 명확히 분리하고,
// TicketService / WorkflowService는 컴포넌트로써 도메인 책임만 담당
// 각 AfterEventHandler는 분기 처리 컴포넌트로 배치됨

package com.example.ticketflow;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.context.event.EventListener;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Component;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.web.bind.annotation.*;

import java.util.Map;

/*
 * TicketController - 요청 진입점
 */
@RestController
@RequestMapping("/ticket")
@RequiredArgsConstructor
public class TicketController {
    private final TicketFlowModule ticketFlowModule;

    @PostMapping("/start/{ticketId}")
    public void start(@PathVariable Long ticketId) {
        ticketFlowModule.startTicket(ticketId);
    }

    @PostMapping("/workflow/reject/{workflowId}")
    public void reject(@PathVariable Long workflowId, @RequestParam AfterEventPolicy policy) {
        ticketFlowModule.rejectWorkflow(workflowId, policy);
    }
}

/*
 * TicketFlowModule - 모듈 역할, 도메인 컴포넌트를 조합해 흐름 구성
 */
@Service
@RequiredArgsConstructor
public class TicketFlowModule {
    private final TicketComponent ticketComponent;
    private final WorkflowComponent workflowComponent;
    private final ApplicationEventPublisher eventPublisher;

    @Transactional
    public void startTicket(Long ticketId) {
        ticketComponent.preStart(ticketId);
        ticketComponent.start(ticketId);
        workflowComponent.start(ticketId);
        eventPublisher.publishEvent(new TicketStartedEvent(ticketId));
    }

    @Transactional
    public void rejectWorkflow(Long workflowId, AfterEventPolicy policy) {
        workflowComponent.reject(workflowId);
        eventPublisher.publishEvent(new WorkflowRejectedEvent(workflowId, policy));
    }
}

/*
 * Event 정의
 */
public record WorkflowRejectedEvent(Long workflowId, AfterEventPolicy policy) {}
public record TicketStartedEvent(Long ticketId) {}

/*
 * AfterEventPolicy - 정책 Enum
 */
public enum AfterEventPolicy {
    RESET, PROCEED, ALERT
}

/*
 * WorkflowEventHandler - 이벤트 리스너 기반 전략 적용 + 로깅
 */
@Component
@RequiredArgsConstructor
@Slf4j
public class WorkflowEventHandler {
    private final Map<AfterEventPolicy, AfterEventHandler> handlers;
    private final HistoryLogger historyLogger;

    @Async
    @EventListener
    public void onRejected(WorkflowRejectedEvent event) {
        AfterEventHandler handler = handlers.get(event.policy());
        if (handler != null) {
            handler.handle(event.workflowId());
            historyLogger.log("WORKFLOW_REJECTED", event.workflowId(), event.policy().name());
        } else {
            log.warn("No handler for policy: {}", event.policy());
        }
    }

    @Async
    @EventListener
    public void onStarted(TicketStartedEvent event) {
        historyLogger.log("TICKET_STARTED", event.ticketId(), "STARTED");
    }
}

/* 전략 인터페이스 */
public interface AfterEventHandler {
    void handle(Long workflowId);
}

/* 전략 구현체들 */
@Component
@RequiredArgsConstructor
class ResetHandler implements AfterEventHandler {
    private final TicketComponent ticketComponent;
    @Override
    public void handle(Long workflowId) {
        ticketComponent.resetByWorkflow(workflowId);
    }
}

@Component
@RequiredArgsConstructor
class ProceedHandler implements AfterEventHandler {
    private final WorkflowComponent workflowComponent;
    @Override
    public void handle(Long workflowId) {
        workflowComponent.moveToNext(workflowId);
    }
}

@Component
class AlertHandler implements AfterEventHandler {
    @Override
    public void handle(Long workflowId) {
        System.out.println("[Alert] 워크플로우 문제 발생: " + workflowId);
    }
}

/* 도메인 컴포넌트: 티켓 */
@Component
@RequiredArgsConstructor
public class TicketComponent {
    private final TicketMapper ticketMapper;

    public void preStart(Long ticketId) {
        // ex. 사전 상태 점검 등
    }

    public void start(Long ticketId) {
        ticketMapper.updateStatus(ticketId, "STARTED");
    }

    public void resetByWorkflow(Long workflowId) {
        ticketMapper.resetByWorkflow(workflowId);
    }
}

/* 도메인 컴포넌트: 워크플로우 */
@Component
@RequiredArgsConstructor
public class WorkflowComponent {
    private final WorkflowMapper workflowMapper;

    public void start(Long ticketId) {
        workflowMapper.insertInitialStep(ticketId);
    }

    public void reject(Long workflowId) {
        workflowMapper.updateStatus(workflowId, "REJECTED");
    }

    public void moveToNext(Long workflowId) {
        workflowMapper.moveToNextStep(workflowId);
    }
}

/* 이력 로깅 */
@Component
@Slf4j
public class HistoryLogger {
    public void log(String eventType, Long refId, String detail) {
        log.info("[HISTORY] event={}, refId={}, detail={}", eventType, refId, detail);
    }
}

/* DAO */
@Mapper
public interface TicketMapper {
    void updateStatus(Long ticketId, String status);
    void resetByWorkflow(Long workflowId);
}

@Mapper
public interface WorkflowMapper {
    void insertInitialStep(Long ticketId);
    void updateStatus(Long workflowId, String status);
    void moveToNextStep(Long workflowId);
}

```
