# Istio 트러블슈팅 실습

> 본 장의 개념을 손으로 확인한다. 실습은 개인 GCP K8s 클러스터(kubeadm 기반 3노드) 위에서 진행하지만, 본문 명령은 환경에 무관한 `kubectl`/`istioctl` 범용 명령으로 쓴다. 환경이 바뀌면 `## 실습 환경` 섹션만 갈아 끼우면 된다.


## 실습 환경

현재 실습 대상은 `gcp` 스킬에 등록된 개인 클러스터다.

| 항목 | 값 | 역할 |
|------|-----|-----|
| dev-server | 4 vCPU / 8GB · asia-northeast3-a | Control Plane |
| dev-server-2 | 4 vCPU / 8GB | Worker |
| dev-server-3 | 4 vCPU / 8GB | Worker |
| K8s | kubeadm v1.31.14 | — |
| Istio | istioctl 기준 설치 | 메시 주입 네임스페이스: `default`, `bookinfo` |

접속은 `gcp` 스킬의 SSH 절차를 따른다. 상세는 [`.claude/skills/gcp/SKILL.md`](../../../../.claude/skills/gcp/SKILL.md) 참조.


## 실습 1. proxy-status로 동기화 상태 확인

```bash
# 전체 프록시 동기화 상태 확인
istioctl proxy-status

# 예상 출력:
# NAME                                      CDS        LDS        EDS        RDS
# productpage-v1-xxxx.bookinfo              SYNCED     SYNCED     SYNCED     SYNCED
# reviews-v1-xxxx.bookinfo                  SYNCED     SYNCED     SYNCED     SYNCED
---

# 특정 Pod 상세 동기화 상태
PRODUCTPAGE=$(kubectl get pod -n bookinfo -l app=productpage \
  -o jsonpath='{.items[0].metadata.name}')
istioctl proxy-status ${PRODUCTPAGE}.bookinfo

# describe로 특정 Pod에 적용된 정책 확인
istioctl describe pod ${PRODUCTPAGE} -n bookinfo
```


## 실습 2. 503 NC 오류 재현 및 해결 (DestinationRule 누락)

```bash
# 문제 상황 생성: subset v2 참조하지만 DR에는 v2가 없음
kubectl apply -f - <<EOF
apiVersion: networking.istio.io/v1
kind: VirtualService
metadata:
  name: reviews-broken
  namespace: bookinfo
spec:
  hosts:
  - reviews
  http:
  - route:
    - destination:
        host: reviews
        subset: v2    # v2 subset 참조
EOF

kubectl apply -f - <<EOF
apiVersion: networking.istio.io/v1
kind: DestinationRule
metadata:
  name: reviews-broken-dr
  namespace: bookinfo
spec:
  host: reviews
  subsets:
  - name: v1          # v2가 없음!
    labels:
      version: v1
EOF

# 503 발생 확인
kubectl exec -n bookinfo deploy/productpage -- \
  curl -s -o /dev/null -w "%{http_code}" http://reviews:9080/reviews/1

# Access Log에서 NC 플래그 확인
kubectl logs -n bookinfo -l app=productpage -c istio-proxy | \
  grep '"response_flags":"NC"' | tail -3

# istioctl analyze로 사전 탐지
istioctl analyze -n bookinfo

# 수정: v2 subset 추가
kubectl apply -f - <<EOF
apiVersion: networking.istio.io/v1
kind: DestinationRule
metadata:
  name: reviews-broken-dr
  namespace: bookinfo
spec:
  host: reviews
  subsets:
  - name: v1
    labels:
      version: v1
  - name: v2
    labels:
      version: v2
EOF

# 복구 확인 (200 반환)
kubectl exec -n bookinfo deploy/productpage -- \
  curl -s -o /dev/null -w "%{http_code}" http://reviews:9080/reviews/1

# 정리
kubectl delete virtualservice reviews-broken -n bookinfo
kubectl delete destinationrule reviews-broken-dr -n bookinfo
```


## 실습 3. 504 UT 오류 재현 및 해결 (타임아웃)

```bash
# ratings에 2초 지연 주입
kubectl apply -f - <<EOF
apiVersion: networking.istio.io/v1
kind: VirtualService
metadata:
  name: ratings-delay
  namespace: bookinfo
spec:
  hosts:
  - ratings
  http:
  - fault:
      delay:
        percentage:
          value: 100.0
        fixedDelay: 2s
    route:
    - destination:
        host: ratings
        subset: v1
EOF

# reviews에 1초 타임아웃 설정
kubectl apply -f - <<EOF
apiVersion: networking.istio.io/v1
kind: VirtualService
metadata:
  name: reviews-timeout
  namespace: bookinfo
spec:
  hosts:
  - reviews
  http:
  - route:
    - destination:
        host: reviews
        subset: v1
    timeout: 1s
EOF

# 504 발생 확인
kubectl exec -n bookinfo deploy/productpage -- \
  curl -s -o /dev/null -w "%{http_code}" http://reviews:9080/reviews/1

# Access Log에서 UT 플래그 확인
kubectl logs -n bookinfo -l app=reviews -c istio-proxy | \
  grep '"response_flags":"UT"' | tail -3

# Prometheus에서 실제 p99 레이턴시 측정 후 타임아웃 조정
# histogram_quantile(0.99, sum(rate(istio_request_duration_milliseconds_bucket{destination_workload="reviews"}[5m])) by (le))
---

# 타임아웃을 3초로 조정 (p99 × 1.5 기준)
kubectl patch virtualservice reviews-timeout -n bookinfo \
  --type merge -p '{"spec":{"http":[{"route":[{"destination":{"host":"reviews","subset":"v1"}}],"timeout":"3s"}]}}'

# 정리
kubectl delete virtualservice ratings-delay reviews-timeout -n bookinfo
```


## 실습 4. proxy-config 체인으로 설정 추적

```bash
PRODUCTPAGE=$(kubectl get pod -n bookinfo -l app=productpage \
  -o jsonpath='{.items[0].metadata.name}')

# 1단계: Listener 확인
echo "=== Listeners (port 9080) ==="
istioctl proxy-config listener ${PRODUCTPAGE}.bookinfo --port 9080

# 2단계: Route 확인 (VirtualService 반영 여부)
echo "=== Routes ==="
istioctl proxy-config route ${PRODUCTPAGE}.bookinfo --name 9080

# 3단계: Cluster 확인 (subset 포함)
echo "=== Clusters ==="
istioctl proxy-config cluster ${PRODUCTPAGE}.bookinfo \
  --fqdn reviews.bookinfo.svc.cluster.local

# 4단계: Endpoint 확인 (Pod IP 등록 여부)
echo "=== Endpoints ==="
istioctl proxy-config endpoint ${PRODUCTPAGE}.bookinfo \
  --cluster "outbound|9080|v1|reviews.bookinfo.svc.cluster.local"

# Endpoint가 비어있으면: 해당 subset 레이블을 가진 Pod 확인
kubectl get pods -n bookinfo -l version=v1
```


## 실습 5. Envoy Admin Interface 활용

```bash
REVIEWS_POD=$(kubectl get pod -n bookinfo -l app=reviews \
  -o jsonpath='{.items[0].metadata.name}')

# 포트 포워딩
kubectl port-forward ${REVIEWS_POD} 15000:15000 -n bookinfo &

# 주요 엔드포인트 확인
# 클러스터 목록과 헬스 상태
curl -s localhost:15000/clusters | grep "reviews"

# 서킷 브레이커 overflow 카운터
curl -s localhost:15000/stats | grep "overflow"

# 현재 로그 레벨 확인
curl -s localhost:15000/logging

# debug 레벨로 변경 (문제 재현 시에만)
curl -X POST "localhost:15000/logging?router=debug&http=debug"

# 로그 확인
kubectl logs ${REVIEWS_POD} -c istio-proxy --tail=20 -f

# info 레벨로 복원
curl -X POST "localhost:15000/logging?level=info"

# 포트 포워딩 종료
kill %1
```


## 실습 6. config_dump 분석

```bash
REVIEWS_POD=$(kubectl get pod -n bookinfo -l app=reviews \
  -o jsonpath='{.items[0].metadata.name}')

# config_dump 크기 확인
kubectl exec -n bookinfo ${REVIEWS_POD} -c istio-proxy -- \
  curl -s localhost:15000/config_dump | wc -c

# 특정 클러스터 설정 추출
kubectl exec -n bookinfo ${REVIEWS_POD} -c istio-proxy -- \
  curl -s localhost:15000/config_dump | \
  python3 -c "
import sys, json
data = json.load(sys.stdin)
for config in data['configs']:
    if 'ClustersConfigDump' in config.get('@type', ''):
        clusters = config.get('dynamic_active_clusters', [])
        for c in clusters[:3]:
            print(c['cluster']['name'])
"
```


## 실습 7. Response Flag 기반 진단 (Prometheus)

```bash
# Response Flag별 에러 분포 확인 (Prometheus 포트포워딩 후)
kubectl port-forward -n istio-system svc/prometheus 9090:9090 &

# 브라우저에서 실행할 쿼리:
# 1. Flag별 에러 분포
# sum by (response_flags) (irate(istio_requests_total{response_code=~"5..", destination_service_namespace="bookinfo"}[5m]))
---

# 2. UO 발생률
# sum(irate(istio_requests_total{response_flags="UO"}[5m])) by (destination_workload)
---

# 3. DC 제외 서버측 에러율
# sum(rate(istio_requests_total{response_code=~"5..", response_flags!="DC"}[5m])) by (destination_workload)
---

# 4. 클라이언트 vs 서버 관점 비교
# sum(rate(istio_requests_total{reporter="source", response_code=~"5.."}[5m]))
# sum(rate(istio_requests_total{reporter="destination", response_code=~"5.."}[5m]))
---

kill %1
```


## 실습 8. istioctl analyze CI/CD 통합

```bash
# 로컬 파일 분석 (클러스터 없이)
cat > /tmp/test-vs.yaml << 'EOF'
apiVersion: networking.istio.io/v1
kind: VirtualService
metadata:
  name: test-vs
  namespace: bookinfo
spec:
  hosts:
  - reviews
  http:
  - route:
    - destination:
        host: reviews
        subset: nonexistent-subset  # 존재하지 않는 subset
EOF

# 배포 전 분석 (IST0101 오류 탐지)
istioctl analyze /tmp/test-vs.yaml --use-kube=false

# 실제 클러스터와 함께 분석
istioctl analyze /tmp/test-vs.yaml -n bookinfo

rm /tmp/test-vs.yaml
```


## 체크리스트

- [ ] `proxy-status`에서 SYNCED/STALE/NOT SENT 세 상태의 의미를 설명할 수 있는가
- [ ] DestinationRule subset 누락으로 503 NC를 재현하고, `istioctl analyze`로 사전 탐지했는가
- [ ] Access Log의 `UT` 플래그로 타임아웃 원인을 확인하고, p99 레이턴시 기반으로 타임아웃을 조정했는가
- [ ] Listener → Route → Cluster → Endpoint 순서로 proxy-config 체인을 추적했는가
- [ ] Envoy Admin Interface에서 `overflow` 카운터와 `clusters` 엔드포인트를 확인했는가
- [ ] `reporter="source"` vs `reporter="destination"` 에러율 차이로 문제 위치를 추론했는가
- [ ] `response_flags!="DC"` 필터로 DC를 제외한 서버측 에러율을 계산했는가
